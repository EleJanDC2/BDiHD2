    async function getConnections(departureAirport, arrivalAirport, year, month, day, hour, minute) {
        try {
            const response = await axios.get(`https://api.flightstats.com/flex/connections/rest/v3/json/firstflightin/${departureAirport}/to/${arrivalAirport}/arriving_before/${year}/${month}/${day}/${hour}/${minute}`, headers);
            return response.data;
        } catch (error) {
            console.error('Error making GET request:', error.response ? error.response.data : error.message);
        }
    }
